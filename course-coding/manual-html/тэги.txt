<html> </html> - весь html
<head> </head> - голова html
<body> </body> - тело html
	bgcolor= - цвет фона
	text= - цвет текста
	background= - картинка на фоне (путь к картинке)
	link= - цвет ссылок
	alink= - цвет активных ссылок
	vlink= - цвет посещенных ссылок
	topmargin= -
	leftmargin= -
	marginwidth= -
	marginheight= -
<title> текст </title> - название страницы
<br> - перенос
	clear= - прерывание обтекания картинки
		"left" - слева
		"right" - справа
		"all" - все
<font> текст </font> - шрифт
	color= - цвет
		"#xxxxxx" - код цвета
	size= - размер
		"+4"..."-2" - размеры
	face= - тип шрифта
		"Times"
		"Times New Roman"
		"Arial"
		"Helvetica"
		"Courier"
		"Verdana"
		"Tahoma"
		"Cosmic Sans"
		"Garamond"
<p> текст </p> - параграф
	align= - выравнивание (выключка)
		"center" - по центру
		"left" - по левому краю
		"right" - по правому краю
		"justify" - по ширине
<center> текст </center> - центрирование
<div> текст </div> - тег выравнивания
	align= - выравнивание (выключка)
		"center" - по центру
		"left" - по левому краю
		"right" - по правому краю
		"justify" - по ширине
<H1> текст </H1> - самый крупный заголовок
<H2> текст </H2> - ...
<H3> текст </H3> - ...
<H4> текст </H4> - ...
<H5> текст </H5> - ...
<H6> текст </H6> - самый мелкий заголовок
<b> текст </b> - полужирный текст
<i> текст </i> - наклонный текст (курсив)
<u> текст </u> - подчеркнутый текст
<s> текст </s> - перечеркнутый шрифт
<strike> текст </strike> - перечеркнутый шрифт
<tt> текст </tt> - моноширинный шрифт
<pre> текст </pre> - моноширинный шрифт
<small> текст </small> - уменьшенный шрифт
<big> текст </big> - увеличенный шрифт
<sup> текст </sup> - верхний индекс
<sub> текст </sub> - нижний индекс
<img> - изображение (картинка)
	src= - ресурс
		"my.jpg" - имя файла
		"my/my.jpg" - путь + имя файла
		"../my.jpg" - на каталог выше
		"http://www.homepage.ru/my/my.jpg" - адрес
	usemap= - имя карты связанной с изображением
		"#karta2" - формат записи карты
	align= - выравнивание
		"left" - слева от текста
		"right" - справа от текста
		"bottom"...
		"middle"...
		"top"...
	vspace= - вертикальное расстояние между текстом и рисунком
		"10" - значение в пикселях
	hspace= - горизонтальное расстояние между текстом и рисунком
		"30" - значение в пикселях
	alt= - пояснение к картинке (при наведении)
		"текст" - текст пояснения
	width= - ширина картинки (в пикселях)
	height= - высота картинки (в пикселях)
	border= - рамка (ширина рамки в пикселях)
	bordercolor= - цвет рамки
<a> текст </a> - ссылка
	href= - адрес ссылки
	title= - подсказка для ссылки
	target= - где открыть окно
		"_blank" - открыть в новом окне
<map> </map> - карта
	name= - имя карты
<area> - область
	href= - адрес ссылки
	shape= - форма
		"rect" - прямоугольник
		"circle" - круг
		"poly" - многоугольник
	coords= - координаты точек формы
		"x1,y1,x2,y2" - для прямоугольника
		"x,y,R" - для круга
		"x1,y1,x2,y2,...,xN,yN" - для многоугольника
	alt= - пояснение к области (при наведении)
<table> </table> -таблица
	border= - рамка (вписывается толщина рамки в пикселях)
	bordercolor= - цвет рамки
	cellspacing= - пространство между ячейками
	cellpadding= - поля в ячейках
<tr> </tr> - строка таблицы
<td> текст </td> - столбец таблицы  (ячейка)
	bgcolor= - цвет фона (ячейки, строки, всей таблицы)
	height= - высота ячейки (таблицы, ряда) (пикселях, процентах)
	width= - ширина ячейки (таблицы, ряда) (пикселях, процентах)
	align= - горизонтальное выравнивание
		"left" - слева
		"right" - справа
		"middle" - по центру
	valign= - вертикальное выравнтвание
		"middle" - по центру
		"top" - сверху
		"bottom" - снизу
	colspan= - занимаемое колличество столбцов
	rowspan= - занимаемое колличество строк
	background= - фон ячейки (таблицы, столбца, строки)
<Hr> - линия
	align= - выравнивание
	width= - ширина (в процентах/пикселях)
	size= - толщина
	NoShade - отмена обьемности
	color= - цвет
